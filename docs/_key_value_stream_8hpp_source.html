214 std::string serialize(const ValueT& val) {
215 if constexpr (std::is_same_v<ValueT, std::string>)
216 {
217 return val;
218 }
219 else if constexpr (std::is_same_v<ValueT, std::istream>)
220 {
221 std::string str;
222 val.seekg(0, std::ios::end);
223 auto size = val.tellg();
224 str.resize(size);
225 val.seekg(0);
226 val.read(str.data(), size);
227 return str;
228 }
229 else if constexpr (std::is_same_v<ValueT, Asset>)
230 {
231 std::string str;
232 str.resize(*val.size);
233 memcpy(str.data(), val.ptr, *val.size);
234 return str;
235 }
236 else if constexpr (std::is_trivially_copyable_v<ValueT>)
237 {
238 std::string str;
239 str.resize(sizeof(ValueT));
240 memcpy(str.data(), &val, sizeof(ValueT));

242 }
