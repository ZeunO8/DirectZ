518 template <size_t N>
519 mat& rotate(T angle, const vec<T, N>& axis)
520 {
521 static_assert(N == 2 || N == 3, "Rotation axis must be 2D or 3D");
522 constexpr size_t D = N;
523
524 T c = std::cos(angle);
525 T s = std::sin(angle);
526 T ic = T(1) - c;
527
528 vec<T, 3> a{};
529 if constexpr (N == 2)
530 {
531 a[0] = axis[0];
532 a[1] = axis[1];
533 a[2] = T(0); // implicit Z = 0 for 2D rotation
534 }
535 else
536 {
537 a = axis.normalize();
538 }
539
540 T x = a[0];
541 T y = a[1];
542 T z = a[2];
543
544 // Precompute rotation matrix components
545 T r00, r01, r02;
546 T r10, r11, r12;
547 T r20, r21, r22;
548
549 if constexpr (N == 2)
550 {
551 r00 = c; r01 = -s; r02 = 0;
552 r10 = s; r11 = c; r12 = 0;
553 r20 = 0; r21 = 0; r22 = 1;
554 }
555 else
556 {
557 r00 = x * x * ic + c;
558 r01 = x * y * ic - z * s;
559 r02 = x * z * ic + y * s;
560
561 r10 = y * x * ic + z * s;
562 r11 = y * y * ic + c;
563 r12 = y * z * ic - x * s;
564
565 r20 = z * x * ic - y * s;
566 r21 = z * y * ic + x * s;
567 r22 = z * z * ic + c;
568 }
569
570 // Apply rotation matrix to each column vector
571 for (size_t cidx = 0; cidx < C; ++cidx)
572 {
573 vec<T, R>& col = (vec<T, R>&)(*this)[cidx];
574
575 T vx = (D > 0) ? col[0] : T(0);
576 T vy = (D > 1) ? col[1] : T(0);
577 T vz = (D > 2) ? col[2] : T(0);
578
579 if constexpr (N == 2)
580 {
581 col[0] = vx * r00 + vy * r01;
582 col[1] = vx * r10 + vy * r11;
583 }
584 else
585 {
586 col[0] = vx * r00 + vy * r01 + vz * r02;
587 col[1] = vx * r10 + vy * r11 + vz * r12;
588 col[2] = vx * r20 + vy * r21 + vz * r22;
589 }
590 // Remaining rows (3 and above) remain unchanged
591 }
592
593 return *this;
594 }

636 mat<T, C, R> inverse() const
637 {
638 static_assert(C == R, "Inverse only defined for square matrices");
639 constexpr size_t N = C;
640 mat<T, N, N> result;
641 mat<T, N, N> copy = *this;
642 result = mat<T, N, N>((T)1); // Initialize result to identity
643
644 for (size_t i = 0; i < N; ++i)
645 {
646 T pivot = copy.matrix[i][i];
647 if (std::abs(pivot) < std::numeric_limits<T>::epsilon())
648 {
649 size_t swap_row = i + 1;
650 while (swap_row < N && std::abs(copy.matrix[swap_row][i]) < std::numeric_limits<T>::epsilon())
651 ++swap_row;
652 if (swap_row == N)
653 assert(false && "Matrix is singular and cannot be inverted");
654 for (size_t k = 0; k < N; ++k)
655 {
656 std::swap(copy.matrix[i][k], copy.matrix[swap_row][k]);
657 std::swap(result.matrix[i][k], result.matrix[swap_row][k]);
658 }
659 pivot = copy.matrix[i][i];
660 }
661
662 T inv_pivot = (T)1 / pivot;
663 for (size_t j = 0; j < N; ++j)
664 {
665 copy.matrix[i][j] *= inv_pivot;
666 result.matrix[i][j] *= inv_pivot;
667 }
668
669 for (size_t row = 0; row < N; ++row)
670 {
671 if (row == i)
672 continue;
673 T factor = copy.matrix[row][i];
674 for (size_t col = 0; col < N; ++col)
675 {
676 copy.matrix[row][col] -= factor * copy.matrix[i][col];
677 result.matrix[row][col] -= factor * result.matrix[i][col];
678 }
679 }
680 }
681
682 return result;
683 }

698 mat<T, C, R>& operator*=(const mat<T, C2, R2>& rhs)
699 {
700 static_assert(C == R2, "Matrix multiplication dimension mismatch");

702 // Temporary storage for results
703 T temp[C2][R];
704
705 for (size_t c = 0; c < C2; c++)

709 T sum = T(0);

719 static_assert(C2 == C && R == R2, "operator*= requires output size equals current matrix size");

773 template<typename T>
774 mat<T, 4, 4> lookAt(const vec<T, 3>& eye, const vec<T, 3>& center, const vec<T, 3>& up)
775 {
776 vec<T, 3> f = (center - eye);
777 T f_len = std::sqrt(f[0]*f[0] + f[1]*f[1] + f[2]*f[2]);
778 f /= f_len;
779
780 vec<T, 3> up_n = up;
781 T up_len = std::sqrt(up[0]*up[0] + up[1]*up[1] + up[2]*up[2]);
782 up_n /= up_len;
783
784 vec<T, 3> s = {
785 f[1] * up_n[2] - f[2] * up_n[1],
786 f[2] * up_n[0] - f[0] * up_n[2],
787 f[0] * up_n[1] - f[1] * up_n[0]
788 };
789 T s_len = std::sqrt(s[0]*s[0] + s[1]*s[1] + s[2]*s[2]);
790 s /= s_len;
791
792 vec<T, 3> u = {
793 s[1] * f[2] - s[2] * f[1],
794 s[2] * f[0] - s[0] * f[2],
795 s[0] * f[1] - s[1] * f[0]
796 };
797
798 mat<T, 4, 4> result((T)1);
799 result[0][0] = s[0]; result[1][0] = s[1]; result[2][0] = s[2];
800 result[0][1] = u[0]; result[1][1] = u[1]; result[2][1] = u[2];
801 result[0][2] = -f[0]; result[1][2] = -f[1]; result[2][2] = -f[2];
802 result[3][0] = - (s[0]*eye[0] + s[1]*eye[1] + s[2]*eye[2]);
803 result[3][1] = - (u[0]*eye[0] + u[1]*eye[1] + u[2]*eye[2]);
804 result[3][2] = + (f[0]*eye[0] + f[1]*eye[1] + f[2]*eye[2]);
805 return result;
806 }

810 {
811 assert(std::abs(aspect - std::numeric_limits<T>::epsilon()) > static_cast<T>(0));
