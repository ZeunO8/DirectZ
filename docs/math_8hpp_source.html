218 template <size_t N>
219 mat& rotate(T angle, const vec<T, N>& axis)
220 {
221 static_assert(N == 2 || N == 3, "Rotation axis must be 2D or 3D");
222 constexpr size_t D = N;
223
224 T c = std::cos(angle);
225 T s = std::sin(angle);
226 T ic = T(1) - c;
227
228 vec<T, 3> a{};
229 if constexpr (N == 2)
230 {
231 a[0] = axis[0];
232 a[1] = axis[1];
233 a[2] = T(0); // implicit Z = 0 for 2D rotation
234 }
235 else
236 {
237 a = axis.normalize();
238 }
239
240 T x = a[0];
241 T y = a[1];
242 T z = a[2];
243
244 // Precompute components of rotation (Rodrigues for N==3, planar rot for N==2)
245 T r00, r01, r02;
246 T r10, r11, r12;
247 T r20, r21, r22;
248
249 if constexpr (N == 2)
250 {
251 r00 = c; r01 = -s; r02 = 0;
252 r10 = s; r11 = c; r12 = 0;
253 r20 = 0; r21 = 0; r22 = 1; // identity for unused Z
254 }
255 else
256 {
257 r00 = x * x * ic + c;
258 r01 = x * y * ic - z * s;
259 r02 = x * z * ic + y * s;
260
261 r10 = y * x * ic + z * s;
262 r11 = y * y * ic + c;
263 r12 = y * z * ic - x * s;
264
265 r20 = z * x * ic - y * s;
266 r21 = z * y * ic + x * s;
267 r22 = z * z * ic + c;
268 }
269
270 // In-place rotation: apply to each column in the matrix
271 for (size_t cidx = 0; cidx < C; ++cidx)
272 {
273 vec<T, R>& col = (vec<T, R>&)(*this)[cidx];
274
275 T vx = (D > 0) ? col[0] : T(0);
276 T vy = (D > 1) ? col[1] : T(0);
277 T vz = (D > 2) ? col[2] : T(0);
278
279 if constexpr (N == 2)
280 {
281 col[0] = vx * r00 + vy * r01;
282 col[1] = vx * r10 + vy * r11;
283 }
284 else
285 {
286 col[0] = vx * r00 + vy * r01 + vz * r02;
287 col[1] = vx * r10 + vy * r11 + vz * r12;
288 col[2] = vx * r20 + vy * r21 + vz * r22;
289 }
290 // Leave remaining rows [3..R) untouched
291 }
292
293 return *this;
294 }

311 mat<T, C, R> inverse() const
312 {
313 static_assert(C == R, "Inverse only defined for square matrices");
314 constexpr size_t N = C;
315 mat<T, N, N> result;
316 mat<T, N, N> copy = *this;
317 result = mat<T, N, N>((T)1);
318
319 for (size_t i = 0; i < N; ++i)
320 {
321 T pivot = copy.matrix[i][i];
322 if (std::abs(pivot) < std::numeric_limits<T>::epsilon())
323 {
324 size_t swap_row = i + 1;
325 while (swap_row < N && std::abs(copy.matrix[swap_row][i]) < std::numeric_limits<T>::epsilon())
326 ++swap_row;
327 if (swap_row == N)
328 assert(false && "Matrix is singular and cannot be inverted");
329 for (size_t k = 0; k < N; ++k)
330 {
331 std::swap(copy.matrix[i][k], copy.matrix[swap_row][k]);
332 std::swap(result.matrix[i][k], result.matrix[swap_row][k]);
333 }
334 pivot = copy.matrix[i][i];
335 }
336
337 T inv_pivot = (T)1 / pivot;
338 for (size_t j = 0; j < N; ++j)
339 {
340 copy.matrix[i][j] *= inv_pivot;
341 result.matrix[i][j] *= inv_pivot;
342 }
343
344 for (size_t row = 0; row < N; ++row)
345 {
346 if (row == i)
347 continue;
348 T factor = copy.matrix[row][i];
349 for (size_t col = 0; col < N; ++col)
350 {
351 copy.matrix[row][col] -= factor * copy.matrix[i][col];
352 result.matrix[row][col] -= factor * result.matrix[i][col];
353 }
354 }
355 }
356
357 return result;
358 }
359 template <size_t C2, size_t R2>
360 mat<T, C, R>& operator*=(const mat<T, C2, R2>& rhs)
361 {
362 static_assert(C == R2, "Matrix multiplication dimension mismatch");
363
364 // Allocate fixed-size temporary buffer on stack
365 // result size is C2 columns, R rows
366 T temp[C2][R];
367
368 for (size_t c = 0; c < C2; c++)
369 {
370 for (size_t r = 0; r < R; r++)
371 {
372 T sum = T(0);
373
374 for (size_t k = 0; k < C; k++)
375 {
376 sum += matrix[k][r] * rhs.matrix[c][k];
377 }
378
379 temp[c][r] = sum;
380 }
381 }
382
383 // Now copy the temp buffer back into matrix, assuming dimensions match
384 static_assert(C2 == C && R == R2, "operator*= requires output size equals current matrix size");
385
386 memcpy(matrix, temp, sizeof(temp));
387
388 return *this;
389 }

398 template<typename T>
399 mat<T, 4, 4> lookAt(const vec<T, 3>& eye, const vec<T, 3>& center, const vec<T, 3>& up)
400 {
401 vec<T, 3> f = (center - eye);
402 T f_len = std::sqrt(f[0]*f[0] + f[1]*f[1] + f[2]*f[2]);
403 f /= f_len;
404
405 vec<T, 3> up_n = up;
406 T up_len = std::sqrt(up[0]*up[0] + up[1]*up[1] + up[2]*up[2]);
407 up_n /= up_len;
408
409 vec<T, 3> s = {
410 f[1] * up_n[2] - f[2] * up_n[1],
411 f[2] * up_n[0] - f[0] * up_n[2],
412 f[0] * up_n[1] - f[1] * up_n[0]
413 };
414 T s_len = std::sqrt(s[0]*s[0] + s[1]*s[1] + s[2]*s[2]);
415 s /= s_len;
416
417 vec<T, 3> u = {
418 s[1] * f[2] - s[2] * f[1],
419 s[2] * f[0] - s[0] * f[2],
420 s[0] * f[1] - s[1] * f[0]
421 };
422
423 mat<T, 4, 4> result((T)1);
424 result[0][0] = s[0]; result[1][0] = s[1]; result[2][0] = s[2];
425 result[0][1] = u[0]; result[1][1] = u[1]; result[2][1] = u[2];
426 result[0][2] = -f[0]; result[1][2] = -f[1]; result[2][2] = -f[2];
427 result[3][0] = - (s[0]*eye[0] + s[1]*eye[1] + s[2]*eye[2]);
428 result[3][1] = - (u[0]*eye[0] + u[1]*eye[1] + u[2]*eye[2]);
429 result[3][2] = + (f[0]*eye[0] + f[1]*eye[1] + f[2]*eye[2]);
430 return result;
431 }

628 inline static void reseed_mt19937_with_current_time()
629 {

631 auto duration = now.time_since_epoch();
632 auto nanos = std::chrono::duration_cast<std::chrono::nanoseconds>(duration).count();
633 std::uint32_t new_seed = static_cast<std::uint32_t>(nanos & 0xFFFFFFFF);
634 _mt19937.seed(new_seed);

638 template <typename T>
639 static const T value(const T min, const T max, const size_t seed = (std::numeric_limits<size_t>::max)())
640 {
641 std::mt19937* mt19937Pointer = 0;
642 if (seed != (std::numeric_limits<size_t>::max)())
643 {
644 mt19937Pointer = new std::mt19937(seed);
645 }
646 else
647 {
648 reseed_mt19937_with_current_time();

654 auto value = distrib(*mt19937Pointer);
655 if (seed != (std::numeric_limits<size_t>::max)())
656 {
657 delete mt19937Pointer;
658 mt19937Pointer = 0;
659 }
660 return value;
661 }
662 else if constexpr (std::is_integral<T>::value)
663 {
664 std::uniform_int_distribution<T> distrib(min, max);
665 auto value = distrib(*mt19937Pointer);
666 if (seed != (std::numeric_limits<size_t>::max)()) // ∞
667 {
668 delete mt19937Pointer;
669 mt19937Pointer = 0;
670 }
671 return value;
672 }
673 throw std::runtime_error("Type is not supported by Random::value");
