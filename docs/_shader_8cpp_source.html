79void ReflectedStructView::set_member(const std::string& member_name, const void* data_ptr, size_t data_size_bytes) {
80 auto offset_it = m_struct_def.member_offsets_map.find(member_name);
81 auto size_it = m_struct_def.member_sizes_map.find(member_name);
82
83 if (offset_it == m_struct_def.member_offsets_map.end() ||
84 size_it == m_struct_def.member_sizes_map.end()) {
85 throw std::runtime_error("ReflectedStructView Error: Member '" + member_name + "' not found in struct '"
86 + m_struct_def.name + "'.");
87 }
88
89 uint32_t member_offset = offset_it->second;
90 uint32_t reflected_size = size_it->second;
91
92 if (!(reflected_size == data_size_bytes))
93 {
94 throw std::runtime_error("ReflectedSize does not match data_size (bytes)");
95 }
96
97 uint8_t* target_addr = m_base_ptr + member_offset;
98 size_t bytes_to_copy = std::min(data_size_bytes, (size_t)reflected_size);
99
100 memcpy(target_addr, data_ptr, bytes_to_copy);
101}

255std::string GetTypeKindString(const SpvReflectTypeDescription& type_desc) {
256 std::string kind_str;
257 if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT) != 0) {
258 kind_str = "struct";
259 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_ARRAY) != 0) {
260 kind_str = "array";
261 // Use struct_type_description for array element type
262 if (type_desc.struct_type_description) {
263 kind_str += " of " + GetTypeKindString(*type_desc.struct_type_description);
264 }
265 if (type_desc.traits.array.dims_count > 0) {
266 kind_str += " [";
267 for (uint32_t i = 0; i < type_desc.traits.array.dims_count; ++i) {
268 if (i > 0) kind_str += "x";
269 if (type_desc.traits.array.dims[i] == 0) { // Runtime array
270 kind_str += "runtime";
271 } else {
272 kind_str += std::to_string(type_desc.traits.array.dims[i]);
273 }
274 }
275 kind_str += "]";
276 }
277 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_MATRIX) != 0) {
278 kind_str = "mat" + std::to_string(type_desc.traits.numeric.matrix.column_count) + "x" + std::to_string(type_desc.traits.numeric.matrix.row_count);
279 // Use struct_type_description for base scalar type of matrix components
280 if (type_desc.struct_type_description) {
281 kind_str += " of " + GetTypeKindString(*type_desc.struct_type_description);
282 }
283 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_VECTOR) != 0) {
284 kind_str = "vec" + std::to_string(type_desc.traits.numeric.vector.component_count);
285 // Use struct_type_description for base scalar type of vector components
286 if (type_desc.struct_type_description) {
287 kind_str += " of " + GetTypeKindString(*type_desc.struct_type_description);
288 }
289 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_FLOAT) != 0) {
290 kind_str = "float" + std::to_string(type_desc.traits.numeric.scalar.width);
291 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_INT) != 0) {
292 // Use traits.numeric.scalar.signedness directly
293 kind_str = (type_desc.traits.numeric.scalar.signedness != 0) ? "int" : "uint";
294 kind_str += std::to_string(type_desc.traits.numeric.scalar.width);
295 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_BOOL) != 0) {
296 kind_str = "bool";
297 } else if (type_desc.op == SpvOpTypeImage) {
298 kind_str = "image";
299 } else if (type_desc.op == SpvOpTypeSampler) {
300 kind_str = "sampler";
301 } else if (type_desc.op == SpvOpTypeSampledImage) {
302 kind_str = "sampled_image";
303 } else if (type_desc.op == SpvOpTypeVoid) {
304 kind_str = "void";
305 } else {
306 kind_str = "unknown_op_" + std::to_string(type_desc.op);
307 }
308 return kind_str;
309}

323uint32_t GetMinimumTypeSizeInBytes(const SpvReflectTypeDescription& type_desc) {
324 // SpvReflectTypeDescription itself doesn't directly expose 'size'.
325 // We derive it from traits for basic types.
326 // For structs, the 'size' field typically exists on SpvReflectBlockVariable,
327 // which describes an instance of the struct, not the definition itself.
328 // We'll return 0 for structs here as their size is context-dependent (padding/alignment).
329
330 if (type_desc.op == SpvOpTypeRuntimeArray) {
331 if (type_desc.struct_member_name)
332 {
333 return CalculateStructSize(type_desc);
334 }
335 return 0; // Runtime arrays have size determined at runtime
336 }
337
338 if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_ARRAY) != 0) {
339 // Use struct_type_description for array element type
340 if (type_desc.traits.array.dims_count > 0) {
341 uint32_t total_elements = 1;
342 for (uint32_t i = 0; i < type_desc.traits.array.dims_count; ++i) {
343 if (type_desc.traits.array.dims[i] == 0) return 0;
344 total_elements *= type_desc.traits.array.dims[i];
345 }
346 return total_elements * (type_desc.traits.array.stride ? type_desc.traits.array.stride : (type_desc.traits.numeric.scalar.width / 8));
347 }
348 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_MATRIX) != 0) {
349 return type_desc.traits.numeric.matrix.column_count * type_desc.traits.numeric.matrix.row_count * (type_desc.traits.numeric.scalar.width / 8);
350 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_VECTOR) != 0) {
351 return type_desc.traits.numeric.vector.component_count * (type_desc.traits.numeric.scalar.width / 8);
352 } else if ((type_desc.type_flags & (SPV_REFLECT_TYPE_FLAG_FLOAT | SPV_REFLECT_TYPE_FLAG_INT | SPV_REFLECT_TYPE_FLAG_BOOL)) != 0) {
353 return type_desc.traits.numeric.scalar.width / 8; // Convert bits to bytes
354 } else if ((type_desc.type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT) != 0) {
355 return CalculateStructSize(type_desc);
356 }
357 return 0; // Default if size cannot be determined for other types
358}

373size_t HashStructSignature(const SpvReflectTypeDescription& type)
374{
375 size_t hash = 0;
376 for (uint32_t i = 0; i < type.member_count; ++i)
377 {
378 const auto& m = type.members[i];
379 hash ^= std::hash<std::string>()(m.struct_member_name ? m.struct_member_name : "") << 1;
380 hash ^= m.type_flags + 0x9e3779b9;
381 hash ^= m.traits.numeric.scalar.width; // Use width as a basic signature component
382 // Note: m.offset is part of SpvReflectBlockVariable, not SpvReflectTypeDescription member.
383 // If it's a direct member of a struct, you might not have offset here.
384 // For `SpvReflectTypeDescription::members`, the 'members' field is deprecated
385 // and doesn't contain a 'member_variable' field like SpvReflectBlockVariable.
386 // Let's remove m.offset from hash signature as it's not directly available on SpvReflectTypeDescription's members.
387 }
388 return hash;
389}

391void ReflectAllTypes(const SpvReflectShaderModule& module, SPIRVReflection& out)
392{
393 auto& internal = *module._internal;
394 std::unordered_map<size_t, uint32_t> seen_structs;
395
396 for (uint32_t i = 0; i < internal.type_description_count; ++i)
397 {
398 const SpvReflectTypeDescription& type = internal.type_descriptions[i];
399 // Only process named types or structs that might be referenced
400 if (!(type.type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT))
401 continue;
402
403 ReflectedType rtype = CreateReflectedType(type);
404 if (!rtype.name.empty()) { // Only add named types to the map
405 out.named_types[rtype.name] = rtype;
406 }
407
408 if ((type.type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT) != 0)
409 {
410 size_t sig = HashStructSignature(type);
411 if (seen_structs.find(sig) == seen_structs.end())
412 {
413 seen_structs[sig] = type.id;
414 ReflectedStruct s(rtype.name, rtype.id, type);
415 out.structs_by_name[rtype.name] = s;
416 out.structs_by_id[rtype.id] = s;
417 }
418 }
419 }

422void ReflectStructMembersRecursive(const SpvReflectTypeDescription* type_desc, std::vector<ReflectedVariable>& container, const std::string& prefix = "")
423{
424 if (!type_desc || !(type_desc->type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT))
425 return;
426
427 // Use the type_desc->members (which are SpvReflectTypeDescription) for recursive calls
428 // but the actual block variable members (if available) for offsets/strides
429 for (uint32_t mi = 0; mi < type_desc->member_count; ++mi)
430 {
431 const SpvReflectTypeDescription* member_type_desc = &type_desc->members[mi];
432 // Note: SpvReflectTypeDescription::members are just type descriptions,
433 // they don't have the full SpvReflectBlockVariable info (like offset, array.stride, decorations)
434 // unless they are explicitly block variables themselves.
435 // For structs, member_variable isn't directly exposed on SpvReflectTypeDescription members.
436 // These fields are typically available when iterating through SpvReflectBlockVariable::members.
437
438 ReflectedVariable rv;
439 rv.name = prefix + (member_type_desc->struct_member_name ? member_type_desc->struct_member_name : "");
440 // Offsets, strides, and decorations for struct members are typically found in the SpvReflectBlockVariable
441 // that *contains* this struct, not on the struct member's type description itself.
442 // For now, these will be default values here unless passed explicitly or looked up.
443 // If this function is only called for actual block members (UBO/SSBO), then the `member_block_var` is valid.
444 // However, the current signature passes type_desc directly.
445
446 // To get accurate offsets/strides/decorations for nested members,
447 // this function would ideally be called with the actual SpvReflectBlockVariable members
448 // if reflecting a uniform/storage buffer.
449 // For now, we'll populate basic type info.
450 rv.type = CreateReflectedType(*member_type_desc);
451 // rv.offset and rv.array_stride cannot be reliably obtained from SpvReflectTypeDescription::members alone.
452 // rv.decorations will be based on the type's decorations, not the variable's.
453
454
455 if (member_type_desc->type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT)
456 {
457 // Recursively add members of nested structs
458 ReflectStructMembersRecursive(member_type_desc, container, rv.name + ".");
459 }
460 else if ((member_type_desc->type_flags & SPV_REFLECT_TYPE_FLAG_ARRAY) && member_type_desc->struct_type_description)
461 {
462 // If it's an array of structs, recurse into the element type.
463 if (member_type_desc->struct_type_description->type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT) {
464 ReflectStructMembersRecursive(member_type_desc->struct_type_description, container, rv.name + "[].");
465 } else {
466 container.push_back(rv); // Add array of basic types as a leaf variable
467 }
468 }
469 else
470 {
471 container.push_back(rv); // Add basic type as a leaf variable
472 }
473 }
474}

620void ReflectAndPrepareBuffers(BufferGroup* buffer_group, const SpvReflectShaderModule& module, Shader* shader, SPIRVReflection& reflection)
621{
622 for (uint32_t i = 0; i < module.descriptor_binding_count; ++i)
623 {
624 const auto& binding_info = module.descriptor_bindings[i];
625 std::string name = binding_info.name ? binding_info.name : "";
626
627 if (name.empty()) {
628 std::cerr << "Warning: Skipping unnamed descriptor_binding at set=" << binding_info.set << ", binding=" << binding_info.binding << std::endl;
629 continue;
630 }
631 // If we are restricted to buffers and this is not a restriction, skip.
632 if (buffer_group->restricted_to_keys.size() && (buffer_group->restricted_to_keys.find(name)) == buffer_group->restricted_to_keys.end())
633 continue;
634
635 auto vk_descriptor_type = static_cast<VkDescriptorType>(binding_info.descriptor_type);
636
637 switch (binding_info.descriptor_type)
638 {
639 default:
640 {
641 std::cerr << "Warning: Unhandled descriptor type: " << vk_descriptor_type << std::endl;
642 break;
643 }
644 case SPV_REFLECT_DESCRIPTOR_TYPE_SAMPLER:
645 case SPV_REFLECT_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER:
646 case SPV_REFLECT_DESCRIPTOR_TYPE_SAMPLED_IMAGE:
647 case SPV_REFLECT_DESCRIPTOR_TYPE_STORAGE_IMAGE:
648 {
649 if (buffer_group->images.count(name))
650 {
651 auto& image_data = buffer_group->images[name];
652 image_data.descriptor_types[shader] = vk_descriptor_type;
653 image_data.expected_layouts[shader] = infer_required_image_layout(vk_descriptor_type);
654 std::cout << "Added Descriptor Type: " << vk_descriptor_type << " to Reflected image '" << name << "'" << std::endl;
655 continue;
656 }
657
658
659 ShaderImage image_data{};
660 image_data.name = name;
661 image_data.set = binding_info.set;
662 image_data.binding = binding_info.binding;
663 image_data.descriptor_types[shader] = vk_descriptor_type;
664 image_data.viewType = infer_view_type(binding_info.image);
665 image_data.format = infer_format(binding_info.image.image_format);
666 image_data.expected_layouts[shader] = infer_required_image_layout(vk_descriptor_type);
667
668 buffer_group->images[name] = image_data;
669 std::cout << "Reflected image '" << name << "' (Set=" << image_data.set << ", Binding=" << image_data.binding
670 << ", Descriptor Type=" << vk_descriptor_type << ", View Type=" << image_data.viewType
671 << ", Format=" << image_data.format << ")" << std::endl;
672
673 break;
674 }
675 case SPV_REFLECT_DESCRIPTOR_TYPE_UNIFORM_BUFFER:
676 case SPV_REFLECT_DESCRIPTOR_TYPE_STORAGE_BUFFER:
677 {
678 // If another shader stage already reflected this buffer, skip.
679 if (buffer_group->buffers.count(name)) {
680 continue;
681 }
682
683 ShaderBuffer buffer_data{};
684 buffer_data.name = name;
685 buffer_data.set = binding_info.set;
686 buffer_data.binding = binding_info.binding;
687 buffer_data.descriptor_type = vk_descriptor_type;
688 buffer_data.static_size = binding_info.block.size; // This is the total size of the block from reflection
689
690 // Determine element_type and related properties
691 const auto& block = binding_info.block; // The entire block (struct or wrapper for content)
692 const SpvReflectTypeDescription* element_type_desc_ptr = nullptr; // Pointer to the type description of one element
693
694 if (binding_info.descriptor_type == SPV_REFLECT_DESCRIPTOR_TYPE_UNIFORM_BUFFER) {
695 // For UBOs, the "element" is the entire block struct itself.
696 element_type_desc_ptr = block.type_description;
697 buffer_data.element_stride = block.size; // Stride is the size of the whole UBO struct
698 buffer_data.element_count = 1; // UBOs typically have one instance
699 buffer_data.is_dynamic_sized = false; // UBOs are fixed size
700 std::cout << "Detected UBO '" << name << "': single struct element." << std::endl;
701 } else if (binding_info.descriptor_type == SPV_REFLECT_DESCRIPTOR_TYPE_STORAGE_BUFFER) {
702 // For SSBOs, we need to look into its members for the actual element type,
703 // especially if it's an array. SSBOs are typically structs containing arrays.
704 if (block.member_count > 0) {
705 const auto& primary_member = block.members[0]; // Assume the first member holds the data structure we care about
706
707 if (primary_member.type_description->op == SpvOpTypeRuntimeArray) {
708 // This is an SSBO with a runtime array as its primary content (e.g., `buffer MyBlock { SomeType data[]; }`)
709 element_type_desc_ptr = primary_member.type_description;
710 buffer_data.is_dynamic_sized = true;
711 if (element_type_desc_ptr->struct_member_name)
712 {
713 buffer_data.element_stride = GetMinimumTypeSizeInBytes(*primary_member.type_description);
714 }
715 else
716 {
717 buffer_data.element_stride = primary_member.type_description->traits.array.stride;
718 }
719 buffer_data.element_count = 0; // Requires application to set size for dynamic buffers
720 std::cout << "Detected dynamic SSBO '" << name << "': runtime array of " << element_type_desc_ptr->type_name << "s." << std::endl;
721 } else if (primary_member.type_description->op == SpvOpTypeArray) {
722 // This is an SSBO with a fixed-size array as its primary content (e.g., `buffer MyBlock { SomeType data[10]; }`)
723 element_type_desc_ptr = primary_member.type_description;
724 buffer_data.is_dynamic_sized = false; // Fixed size
725 buffer_data.element_stride = primary_member.type_description->traits.array.stride;
726 // buffer_data.element_count = primary_member.type_description->traits.array.length; // Explicit length
727 std::cout << "Detected fixed-size array SSBO '" << name << "': array of " << element_type_desc_ptr->type_name << "s with count " << buffer_data.element_count << "." << std::endl;
728 } else {
729 // This is an SSBO that contains a single struct or primitive as its primary member
730 // (e.g., `buffer MyBlock { MyStruct data; }` or `buffer MyBlock { float value; }`)
731 element_type_desc_ptr = primary_member.type_description;
732 buffer_data.is_dynamic_sized = false;
733 buffer_data.element_stride = primary_member.size; // Stride is simply the size of that single member
734 buffer_data.element_count = 1;
735 std::cout << "Detected single-element SSBO '" << name << "': element type " << element_type_desc_ptr->type_name << "." << std::endl;
736 }
737 } else {
738 // This is an SSBO block with no members. This is unusual for data buffers.
739 // It might be a simple SSBO with just a type binding_info.type_description
740 // if it's not a block, but here we are in the 'block' context.
741 std::cerr << "Warning: SSBO '" << name << "' block has no members. Cannot reliably determine element type." << std::endl;
742 // Fallback: If it's a direct binding to a non-struct type, use that.
743 if (!(binding_info.type_description->type_flags & SPV_REFLECT_TYPE_FLAG_STRUCT) && binding_info.type_description->op != SpvOpTypeStruct) {
744 element_type_desc_ptr = binding_info.type_description;
745 buffer_data.is_dynamic_sized = false;
746 buffer_data.element_stride = GetMinimumTypeSizeInBytes(*binding_info.type_description);
747 buffer_data.element_count = 1;
748 std::cout << "Detected simple (non-struct) SSBO '" << name << "' via binding_info.type_description." << std::endl;
749 } else {
750 std::cerr << "Error: Skipping SSBO '" << name << "' due to ambiguous element type." << std::endl;
751 continue;
752 }
753 }
754 }
755
756 // Assign the reflected element type
757 if (element_type_desc_ptr) {
758 buffer_data.element_type = CreateReflectedType(*element_type_desc_ptr);
759
760 // Populate the ReflectedStruct into the shader's reflection if the element type is a struct
761 if (buffer_data.element_type.type_kind == "struct") {
762 // Check if it's already there to avoid duplicates across shader stages
763 if (reflection.structs_by_name.find(buffer_data.element_type.name) == reflection.structs_by_name.end()) {
764 // Create a ReflectedStruct using the type_desc of the element
765 auto s = ReflectedStruct(buffer_data.element_type.name, buffer_data.element_type.id, *element_type_desc_ptr);
766 reflection.structs_by_name.emplace(buffer_data.element_type.name, s);
767 reflection.structs_by_id.emplace(buffer_data.element_type.id, s);
768 std::cout << "Added reflected struct '" << buffer_data.element_type.name << "' (ID: " << buffer_data.element_type.id << ") to global reflection data." << std::endl;
769 }
770 }
771 } else {
772 std::cerr << "Error: Could not determine element type for buffer '" << name << "'." << std::endl;
773 continue; // Skip this buffer if element type can't be determined
774 }
775
776 buffer_group->buffers[name] = buffer_data;
777 std::cout << "Reflected buffer '" << name << "' (Set=" << buffer_data.set << ", Binding=" << buffer_data.binding
778 << ", Static Size=" << buffer_data.static_size << ", Element Stride=" << buffer_data.element_stride
779 << ", Element Type=" << buffer_data.element_type.name << ", Dynamic=" << (buffer_data.is_dynamic_sized ? "Yes" : "No") << ")" << std::endl;
780 break;
781 }
782 }
783 }

789 {

791 ReflectedBlock block;

794 block.set = 0; // in the same way as descriptor sets

796
797 for (uint32_t j = 0; j < push.member_count; ++j)
798 {
799 const SpvReflectBlockVariable& member = push.members[j];
800 ReflectedVariable m;
801 m.name = member.name ? member.name : "";
802 m.offset = member.offset;
803 m.array_stride = member.array.stride;
804 m.type = CreateReflectedType(*member.type_description);
805 // m.decorations = member.decorations;
806 block.members.push_back(m);
807 }
808 out.push_constants.push_back(block); // Store push constants in their own vector
809 }

1142bool CreateDescriptorPool(VkDevice device, Shader* shader, uint32_t max_sets_per_pool)
1143{
1144 std::map<VkDescriptorType, uint32_t> descriptor_counts;
1145
1146 // 1. Aggregate descriptor counts from all shader modules
1147 for (auto const& [stage, module] : shader->module_map)
1148 {
1149 SpvReflectShaderModule* reflect_module = module.reflection.module_ptr.get();
1150 for (uint32_t i = 0; i < reflect_module->descriptor_binding_count; ++i)
1151 {
1152 const SpvReflectDescriptorBinding& binding_info = reflect_module->descriptor_bindings[i];
1153 VkDescriptorType type = static_cast<VkDescriptorType>(binding_info.descriptor_type);
1154 descriptor_counts[type] += binding_info.count;
1155 }
1156 }
1157
1158 if (descriptor_counts.empty()) {
1159 std::cout << "No descriptors found in shader, no pool to create." << std::endl;
1160 return true; // Not an error, just nothing to do.
1161 }
1162
1163 // 2. Create VkDescriptorPoolSize structures
1164 std::vector<VkDescriptorPoolSize> pool_sizes;
1165 for (auto const& [type, count] : descriptor_counts)
1166 {
1167 // We multiply by max_sets_per_pool to allow for multiple sets of this type to be allocated.
1168 pool_sizes.push_back({type, count * max_sets_per_pool});
1169 }
1170
1171 // 3. Create the descriptor pool
1172 VkDescriptorPoolCreateInfo pool_info{};
1173 pool_info.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
1174 pool_info.poolSizeCount = static_cast<uint32_t>(pool_sizes.size());
1175 pool_info.pPoolSizes = pool_sizes.data();
1176 pool_info.maxSets = max_sets_per_pool * static_cast<uint32_t>(shader->descriptor_set_layouts.size());
1177 // You might want VK_DESCRIPTOR_POOL_CREATE_FREE_DESCRIPTOR_SET_BIT if you need to free individual sets
1178 pool_info.flags = 0;
1179
1180 if (vkCreateDescriptorPool(device, &pool_info, nullptr, &shader->descriptor_pool) != VK_SUCCESS)
1181 {
1182 std::cerr << "Failed to create descriptor pool." << std::endl;
1183 return false;
1184 }
1185
1186 std::cout << "Successfully created descriptor pool." << std::endl;
1187 return true;
1188}

1238bool CreateAndBindShaderBuffers(BufferGroup* buffer_group, Shader* shader)
1239{
1240 auto direct_registry = DZ_RGY.get();
1241 std::vector<VkWriteDescriptorSet> descriptor_writes;
1242 std::vector<VkDescriptorBufferInfo> buffer_infos;
1243 std::vector<VkDescriptorImageInfo> image_infos;
1244
1245 for (auto& [name, buffer] : buffer_group->buffers) {
1246 if (buffer.gpu_buffer.mapped_memory)
1247 continue;
1248
1249 VkDeviceSize buffer_size = 0;
1250 if (buffer.is_dynamic_sized) {
1251 if (buffer.element_count == 0) {
1252 std::cout << "Skipping dynamic buffer '" << name << "' as its element count was not set by the application." << std::endl;
1253 continue;
1254 }
1255 buffer_size = buffer.element_count * buffer.element_stride;
1256 } else {
1257 buffer_size = buffer.static_size;
1258 }
1259
1260 if (buffer_size == 0) {
1261 std::cerr << "Warning: Skipping buffer '" << name << "' with zero size." << std::endl;
1262 continue;
1263 }
1264
1265 // Create the GpuBuffer
1266 VkBufferUsageFlags usage = (buffer.descriptor_type == VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER)
1267 ? VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT
1268 : VK_BUFFER_USAGE_STORAGE_BUFFER_BIT;
1269
1270 VkBufferCreateInfo buffer_info{VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO};
1271 buffer_info.size = buffer_size;
1272 buffer_info.usage = usage;
1273 buffer_info.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
1274
1275 if (vkCreateBuffer(direct_registry->device, &buffer_info, nullptr, &buffer.gpu_buffer.buffer) != VK_SUCCESS) {
1276 throw std::runtime_error("Failed to create buffer for " + name);
1277 }
1278
1279 VkMemoryRequirements mem_reqs;
1280 vkGetBufferMemoryRequirements(direct_registry->device, buffer.gpu_buffer.buffer, &mem_reqs);
1281
1282 VkMemoryAllocateInfo alloc_info{VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO};
1283 alloc_info.allocationSize = mem_reqs.size;
1284 alloc_info.memoryTypeIndex = FindMemoryType(direct_registry->physicalDevice, mem_reqs.memoryTypeBits,
1285 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_CACHED_BIT);
1286
1287 if (vkAllocateMemory(direct_registry->device, &alloc_info, nullptr, &buffer.gpu_buffer.memory) != VK_SUCCESS) {
1288 throw std::runtime_error("Failed to allocate memory for buffer " + name);
1289 }
1290 vkBindBufferMemory(direct_registry->device, buffer.gpu_buffer.buffer, buffer.gpu_buffer.memory, 0);
1291
1292 // Persistently map the memory
1293 vkMapMemory(direct_registry->device, buffer.gpu_buffer.memory, 0, buffer_size, 0, &buffer.gpu_buffer.mapped_memory);
1294 buffer.gpu_buffer.size = buffer_size;
1295
1296 // --- Data Upload and Pointer Swap ---
1297 // Ensure data_ptr exists for fixed-size buffers if user wants to get it later
1298 if (!buffer.data_ptr) {
1299 buffer.data_ptr = buffer_group_get_buffer_data_ptr(buffer_group, name);
1300 }
1301
1302 // Copy from CPU staging pointer to mapped GPU pointer
1303 if (buffer.data_ptr) {
1304 memcpy(buffer.gpu_buffer.mapped_memory, buffer.data_ptr.get(), buffer_size);
1305 std::cout << "Copied initial data to GPU for buffer '" << name << "'." << std::endl;
1306 }
1307
1308 // ** THE MAGIC **
1309 // Reset the shared_ptr to point to the GPU mapped memory.
1310 // The custom empty deleter prevents the shared_ptr from trying to free Vulkan's memory.
1311 buffer.data_ptr.reset(static_cast<uint8_t*>(buffer.gpu_buffer.mapped_memory), [](uint8_t*){ /* Do nothing */ });
1312 std::cout << "Remapped data_ptr for '" << name << "' to point directly to GPU memory." << std::endl;
1313 }
1314 for (auto& [name, buffer] : buffer_group->buffers) {
1315 // Prepare the descriptor set write
1316 buffer_infos.emplace_back();
1317 buffer_infos.back().buffer = buffer.gpu_buffer.buffer;
1318 buffer_infos.back().offset = 0;
1319 buffer_infos.back().range = buffer.gpu_buffer.size;
1320 }
1321
1322 size_t i = 0;
1323 for (auto& [name, buffer] : buffer_group->buffers) {
1324 VkWriteDescriptorSet write_set{VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET};
1325 write_set.dstSet = shader->descriptor_sets.at(buffer.set);
1326 write_set.dstBinding = buffer.binding;
1327 write_set.dstArrayElement = 0;
1328 write_set.descriptorType = buffer.descriptor_type;
1329 write_set.descriptorCount = 1;
1330 write_set.pBufferInfo = &buffer_infos[i];
1331 i++;
1332 descriptor_writes.push_back(write_set);
1333 }
1334
1335 for (auto& [name, image_ref] : buffer_group->images) {
1336 auto image_it = buffer_group->runtime_images.find(name);
1337 if (image_it == buffer_group->runtime_images.end())
1338 {
1339 std::cerr << "Warning: Buffer group has not image defined for key: " << name << std::endl;
1340 continue;
1341 }
1342 Image* img = image_it->second.get();
1343 image_infos.emplace_back();
1344 image_infos.back().imageView = img->imageView;
1345 image_infos.back().imageLayout = infer_image_layout(shader, image_ref.descriptor_types);
1346 image_infos.back().sampler = img->sampler; // Use a real sampler if necessary
1347 }
1348
1349 size_t j = 0;
1350 for (auto& [name, image_ref] : buffer_group->images) {
1351 VkWriteDescriptorSet write_set{VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET};
1352 write_set.dstSet = shader->descriptor_sets.at(image_ref.set);
1353 write_set.dstBinding = image_ref.binding;
1354 write_set.dstArrayElement = 0;
1355 auto type = image_ref.descriptor_types[shader];
1356 write_set.descriptorType = type;
1357 write_set.descriptorCount = 1;
1358 write_set.pImageInfo = &image_infos[j];
1359 j++;
1360 descriptor_writes.push_back(write_set);
1361 }
1362
1363 if (!descriptor_writes.empty()) {
1364 vkUpdateDescriptorSets(direct_registry->device, descriptor_writes.size(), descriptor_writes.data(), 0, nullptr);
1365 }
1366
1367 return true;
1368}

1470size_t find_newline_after_token(const std::string& input, const std::string& token)
1471{
1472 // Find the first occurrence of the token in the string
1473 size_t token_pos = input.find(token);
1474
1475 // If token is not found, return std::string::npos as an indicator of failure
1476 if (token_pos == std::string::npos)
1477 {
1478 return std::string::npos;
1479 }
1480
1481 // Calculate the position after the token ends to start searching for newline from there
1482 size_t search_start = token_pos + token.length();
1483
1484 // Find the first occurrence of newline '\n' after the token position
1485 size_t newline_pos = input.find('\n', search_start);
1486
1487 // Return the index of the newline (or npos if not found)
1488 return newline_pos;
1489}

1491std::string addLineNumbers(const std::string &input)
1492{
1493 std::stringstream inputStream(input);
1494 std::stringstream outputStream;
1495 std::string line;
1496 std::size_t lineIndex = 0;
1497
1498 while (std::getline(inputStream, line))
1499 {
1500 outputStream << std::setw(4) << std::setfill(' ') << lineIndex << ": " << line << "\n";
1501 ++lineIndex;
1502 }
1503
1504 // If input ends with a newline, preserve that
1505 if (!input.empty() && input.back() == '\n' && (input.length() == 1 || input[input.length() - 2] != '\r'))
1506 {
1507 // Remove the last newline and add an empty line with a line number
1508 if (input.back() == '\n' && input[input.length() - 2] != '\n')
1509 {
1510 outputStream << std::setw(4) << std::setfill(' ') << lineIndex << ": \n";
1511 }
1512 }
1513
1514 return outputStream.str();

1628void transition_image_layout(VkImage image, VkImageLayout old_layout, VkImageLayout new_layout)
1629{
1630 auto device = DZ_RGY.get()->device;
1631 auto command_buffer = begin_single_time_commands();
1632
1633 VkImageMemoryBarrier barrier{};
1634 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
1635 barrier.oldLayout = old_layout;
1636 barrier.newLayout = new_layout;
1637 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
1638 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
1639 barrier.image = image;
1640 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
1641 barrier.subresourceRange.baseMipLevel = 0;
1642 barrier.subresourceRange.levelCount = 1;
1643 barrier.subresourceRange.baseArrayLayer = 0;
1644 barrier.subresourceRange.layerCount = 1;
1645
1646 auto it = kLayoutTransitions.find({old_layout, new_layout});
1647 if (it == kLayoutTransitions.end())
1648 {
1649 throw std::runtime_error("Unsupported image layout transition!");
1650 }
1651
1652 const TransitionInfo& info = it->second;
1653 barrier.srcAccessMask = info.srcAccessMask;
1654 barrier.dstAccessMask = info.dstAccessMask;

1656 vkCmdPipelineBarrier(
1657 command_buffer,
1658 info.srcStage,
1659 info.dstStage,
1660 0,
1661 0, nullptr,
1662 0, nullptr,
1663 1, &barrier
1664 );
1665
1666 end_single_time_commands(command_buffer);

1669void shader_ensure_image_layouts(Shader* shader)
1670{
1671 for (auto& bound_buffer_group : shader->bound_buffer_groups)
1672 {
1673 for (auto& [name, shader_image] : bound_buffer_group->images)
1674 {
1675 auto runtime_ite = bound_buffer_group->runtime_images.find(name);
1676 if (runtime_ite == bound_buffer_group->runtime_images.end())
1677 continue;
1678
1679 auto& image = *runtime_ite->second;;
1680
1681 auto exp_lay_ite = shader_image.expected_layouts.find(shader);
1682 if (exp_lay_ite == shader_image.expected_layouts.end())
1683 continue;
1684
1685 auto required = exp_lay_ite->second;
1686
1687 if (image.current_layout != required)
1688 {
1689 transition_image_layout(image.image, image.current_layout, required);
1690 image.current_layout = required;
1691 }
1692 }
1693 }

1738void shader_compile(Shader* shader)
1739{
1740 auto direct_registry = DZ_RGY.get();
1741 auto device = direct_registry->device;
1742
1743 // --- Create Pipeline Layout ---
1744 VkPipelineLayoutCreateInfo pipelineLayoutInfo{};
1745 pipelineLayoutInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_LAYOUT_CREATE_INFO;
1746
1747 std::vector<VkDescriptorSetLayout> layouts;
1748 layouts.reserve(shader->descriptor_set_layouts.size());
1749 for (auto& layout : shader->descriptor_set_layouts)
1750 {
1751 layouts.push_back(layout.second);
1752 }
1753
1754 pipelineLayoutInfo.setLayoutCount = static_cast<uint32_t>(layouts.size());
1755 pipelineLayoutInfo.pSetLayouts = layouts.data();
1756 pipelineLayoutInfo.pushConstantRangeCount = 0;
1757 pipelineLayoutInfo.pPushConstantRanges = nullptr;
1758
1759 if (vkCreatePipelineLayout(device, &pipelineLayoutInfo, nullptr, &shader->pipeline_layout) != VK_SUCCESS)
1760 {
1761 throw std::runtime_error("Failed to create pipeline layout!");
1762 }
1763
1764 std::cout << "Created pipeline layout: " << shader->pipeline_layout << std::endl;
1765
1766 // --- Determine Shader Stages ---
1767 std::vector<VkPipelineShaderStageCreateInfo> shaderStages;
1768 bool requires_rasterization = false;
1769 bool uses_compute = false;
1770
1771 for (auto& [module_type, shader_module] : shader->module_map)
1772 {
1773 VkPipelineShaderStageCreateInfo stage_info{};
1774 stage_info.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
1775 stage_info.stage = stageFlags[shader_module.type];
1776 stage_info.module = shader_module.vk_module;
1777 stage_info.pName = "main";
1778
1779 switch (stage_info.stage)
1780 {
1781 case VK_SHADER_STAGE_VERTEX_BIT:
1782 case VK_SHADER_STAGE_FRAGMENT_BIT:
1783 requires_rasterization = true;
1784 break;
1785 case VK_SHADER_STAGE_COMPUTE_BIT:
1786 uses_compute = true;
1787 break;
1788 default:
1789 break;
1790 }
1791
1792 shaderStages.push_back(stage_info);
1793 }
1794
1795 if (uses_compute && requires_rasterization)
1796 {
1797 throw std::runtime_error("Cannot mix compute and raster shader stages in one pipeline");
1798 }
1799
1800 // --- Create Compute Pipeline ---
1801 if (uses_compute)
1802 {
1803 if (shaderStages.size() != 1 || shaderStages[0].stage != VK_SHADER_STAGE_COMPUTE_BIT)
1804 {
1805 throw std::runtime_error("Compute pipelines must have exactly one compute shader stage");
1806 }
1807
1808 VkComputePipelineCreateInfo computeInfo{};
1809 computeInfo.sType = VK_STRUCTURE_TYPE_COMPUTE_PIPELINE_CREATE_INFO;
1810 computeInfo.stage = shaderStages[0];
1811 computeInfo.layout = shader->pipeline_layout;
1812 computeInfo.basePipelineHandle = VK_NULL_HANDLE;
1813 computeInfo.basePipelineIndex = -1;
1814
1815 if (vkCreateComputePipelines(device, VK_NULL_HANDLE, 1, &computeInfo, nullptr, &shader->graphics_pipeline) != VK_SUCCESS)
1816 {
1817 throw std::runtime_error("Failed to create compute pipeline!");
1818 }
1819
1820 std::cout << "Created compute pipeline: " << shader->graphics_pipeline << std::endl;
1821 return;
1822 }

1845 VkPipelineViewportStateCreateInfo viewportState{};

1847 viewportState.viewportCount = 1;
1848 viewportState.pViewports = &dummyViewport;
1849 viewportState.scissorCount = 1;

1854 VK_DYNAMIC_STATE_VIEWPORT,

1856 };
1857
1858 VkPipelineDynamicStateCreateInfo dynamicState{};

1950 vkCmdSetViewport(*direct_registry->commandBuffer, 0, 1, &window.viewport);
1951 vkCmdSetScissor(*direct_registry->commandBuffer, 0, 1, &window.scissor);

1962 auto& draw_list = shader_pair.second;
1963 shader_ensure_image_layouts(shader);
1964 shader_bind(shader);

1966 }
1967 }
1968 }
1969 post_render_pass(renderer);
1970 swap_buffers(renderer);

1973void renderer_draw_commands(Renderer* renderer, Shader* shader, const std::vector<DrawIndirectCommand>& commands)
1974{
1975 auto direct_registry = DZ_RGY.get();
1976 auto drawsSize = commands.size();
1977 auto drawBufferSize = sizeof(VkDrawIndirectCommand) * drawsSize;
1978 auto buffer_hash = drawBufferSize ^ size_t(shader) << 2;
1979 VkCommandBuffer passCB = *direct_registry->commandBuffer;
1980
1981 auto& drawBufferPair = renderer->drawBuffers[buffer_hash];
1982 if (!drawBufferPair.first)
1983 {
1984 createBuffer(
1985 renderer,
1986 drawBufferSize,
1987 VK_BUFFER_USAGE_INDIRECT_BUFFER_BIT,
1988 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
1989 drawBufferPair.first,
1990 drawBufferPair.second
1991 );
1992 }
1993
1994 auto& countBufferPair = renderer->countBuffers[buffer_hash];
1995 if (!countBufferPair.first)
1996 {
1997 createBuffer(
1998 renderer,
1999 sizeof(uint32_t),
2000 VK_BUFFER_USAGE_INDIRECT_BUFFER_BIT,
2001 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
2002 countBufferPair.first,
2003 countBufferPair.second
2004 );
2005 }
2006
2007 std::vector<VkDrawIndirectCommand> i_commands(drawsSize, VkDrawIndirectCommand{});
2008 memcpy(i_commands.data(), commands.data(), i_commands.size() * sizeof(VkDrawIndirectCommand));
2009
2010 // Write draw commands to GPU-visible indirect buffer
2011 void* drawBufferData;
2012 vkMapMemory(direct_registry->device, drawBufferPair.second, 0, VK_WHOLE_SIZE, 0, &drawBufferData);
2013 memcpy(drawBufferData, i_commands.data(), drawBufferSize);
2014 vkUnmapMemory(direct_registry->device, drawBufferPair.second);
2015
2016 // Write draw count to count buffer
2017 uint32_t drawCount = static_cast<uint32_t>(drawsSize);
2018 void* countBufferData;
2019 vkMapMemory(direct_registry->device, countBufferPair.second, 0, VK_WHOLE_SIZE, 0, &countBufferData);
2020 memcpy(countBufferData, &drawCount, sizeof(uint32_t));
2021 vkUnmapMemory(direct_registry->device, countBufferPair.second);
2022
2023 // Descriptor sets
2024 std::vector<VkDescriptorSet> sets;
2025 sets.reserve(shader->descriptor_sets.size());
2026 for (auto& set_pair : shader->descriptor_sets)
2027 {
2028 sets.push_back(set_pair.second);
2029 }
2030
2031 vkCmdBindDescriptorSets(
2032 passCB,
2033 VK_PIPELINE_BIND_POINT_GRAPHICS,
2034 shader->pipeline_layout,
2035 0,
2036 sets.size(),
2037 sets.data(),
2038 0,
2039 nullptr
2040 );
2041
2042 // Detect whether indirect count is supported
2043 bool supportsIndirectCount = false;//renderer->supportsIndirectCount;
2044
2045 if (supportsIndirectCount)
2046 {
2047 vkCmdDrawIndirectCount(
2048 passCB,
2049 drawBufferPair.first,
2050 0,
2051 countBufferPair.first,
2052 0,
2053 drawCount,
2054 sizeof(VkDrawIndirectCommand)
2055 );
2056 }
2057 else
2058 {
2059 // Manually read count from mapped memory
2060 uint32_t fallbackDrawCount = 0;
2061 void* mappedCount = nullptr;
2062 vkMapMemory(direct_registry->device, countBufferPair.second, 0, sizeof(uint32_t), 0, &mappedCount);
2063 memcpy(&fallbackDrawCount, mappedCount, sizeof(uint32_t));
2064 vkUnmapMemory(direct_registry->device, countBufferPair.second);
2065 fallbackDrawCount = std::min(drawCount, fallbackDrawCount);
2066
2067 for (uint32_t i = 0; i < fallbackDrawCount; ++i)
2068 {
2069 VkDeviceSize commandOffset = i * sizeof(VkDrawIndirectCommand);
2070 vkCmdDrawIndirect(passCB, drawBufferPair.first, commandOffset, 1, sizeof(VkDrawIndirectCommand));
2071 }
2072 }
2073}
