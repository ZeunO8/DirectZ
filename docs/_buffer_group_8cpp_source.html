30VkImageUsageFlags infer_image_usage_flags(const std::unordered_map<Shader*, VkDescriptorType>& types)
31{
32 VkImageUsageFlags flags = 0;
33 for (auto& pair : types)
34 {
35 auto& type = pair.second;
36 switch (type)
37 {
38 case VK_DESCRIPTOR_TYPE_SAMPLED_IMAGE:
39 case VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER:
40 flags |= VK_IMAGE_USAGE_SAMPLED_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT;
41
42 case VK_DESCRIPTOR_TYPE_STORAGE_IMAGE:
43 flags |= VK_IMAGE_USAGE_STORAGE_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT;
44
45 case VK_DESCRIPTOR_TYPE_INPUT_ATTACHMENT:
46 flags |= VK_IMAGE_USAGE_INPUT_ATTACHMENT_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT;
47
48 default:
49 flags |= VK_IMAGE_USAGE_SAMPLED_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT; // fallback
50 }
51 }

185ReflectedStructView buffer_group_get_buffer_element_view(BufferGroup* buffer_group, const std::string& buffer_name, uint32_t index)
186{
187 // Find the ShaderBuffer
188 auto it = buffer_group->buffers.find(buffer_name);
189 if (it == buffer_group->buffers.end()) {
190 throw std::runtime_error("shader_get_buffer_element_view: Buffer '" + buffer_name + "' not found.");
191 }
192 ShaderBuffer& buffer = it->second;
193
194 // Ensure data_ptr is valid (allocated CPU-side or mapped GPU-side)
195 if (!buffer.data_ptr) {
196 // If it's a fixed-size buffer and data isn't allocated, attempt to allocate it now.
197 if (!buffer.is_dynamic_sized && buffer.static_size > 0) {
198 buffer.data_ptr = std::shared_ptr<uint8_t>(new uint8_t[buffer.static_size], std::default_delete<uint8_t[]>());
199 std::cout << "Info: Allocating CPU-side buffer for fixed-size buffer '" << buffer_name << "' on first view access." << std::endl;
200 } else {
201 throw std::runtime_error("shader_get_buffer_element_view: Buffer data_ptr is null for '" + buffer_name + "'. "
202 "Ensure shader_set_buffer_element_count() was called for dynamic buffers, or it's mapped/allocated.");
203 }
204 }
205
206 // Validate index for dynamic buffers
207 if (index >= buffer.element_count) {
208 throw std::out_of_range("shader_get_buffer_element_view: Index " + std::to_string(index) +
209 " is out of bounds for buffer '" + buffer_name + "' (element count: " + std::to_string(buffer.element_count) + ").");
210 }
211
212 // Calculate the base address of the desired element within the buffer's data_ptr
213 uint8_t* element_base_ptr = buffer.data_ptr.get() + (index * buffer.element_stride);
214
215 // Get the ReflectedStruct definition for the element type
216 if (buffer.element_type.type_kind != "struct") {
217 throw std::runtime_error("shader_get_buffer_element_view: Buffer '" + buffer_name + "' element is not a struct type. Type kind: " + buffer.element_type.type_kind);
218 }
219
220 for (auto& shader_pair : buffer_group->shaders)
221 {
222 auto shader = shader_pair.first;
223 for (auto& shaderModulePair : shader->module_map)
224 {
225 const ReflectedStruct& reflected_struct_def = getCanonicalStruct(shaderModulePair.second.reflection, buffer.element_type);
226 // Return the ReflectedStructView
227 return ReflectedStructView(element_base_ptr, reflected_struct_def);
228 }
229 }
230 throw std::runtime_error("Could not get ReflectedStruct");
231}
